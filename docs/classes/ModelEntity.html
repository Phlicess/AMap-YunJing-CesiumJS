Class ModelEntity

云境模型实例，继承自 Cesium.Entity

Example

viewer.entities.add(
    new ModelEntity({
        name: 'man',
        position: [lng, lat, height],
        heading,
        scale: 1.2,
        uri: '../data/Cesium_Man.glb',
        eyeOffset: [0, 0, 2],
        viewFrom: [0, -10, 10],
        viewer,
    })
);

    Example

    mdel.moveAlong(
          new AnimationPath([
              {
                  position: [lng, lat, height],
                  time: new Date(),
              },
              {
                  position: [lng + 0.00001, lat, height],
                  time: new Date(new Date().getTime() + 2000),
              },
              {
                  position: [
                      lng + 0.00001,
                      lat + 0.00001,
