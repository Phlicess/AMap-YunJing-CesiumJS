Class SinglePick

智能单体拾取工具

Example

const { SinglePick } = YunJing;
const singlePick = new SinglePick({
    viewer,
    // 云境图层
    layer: yunjing,
    // 单体内点的样式
    inPointStyle: {
        borderWidth: 4,
    },
    // 单体 Mask 的样式
    polygonStyle:{
        color: 'rgba(255, 0, 0, 0.4)',
    },
    model: {
        encoderUrl: '../../onnx/sam_vit_b_01ec64.encoder-fp16.onnx',
        decoderUrl: '../../onnx/sam_vit_b_01ec64.decoder.onnx',
        name: 'test',
        wasmPaths: '../../onnx/'
    }
});
